{{- /* Language switcher — loops hugo.Sites so any N-language site works without template surgery. Shows the OTHER languages (not the current one). Prefers `.Language.Params.label`, then `.Language.Label`, then the bare code. The link target is a translated page if available, otherwise the localised home page. */ -}} {{- $page := . -}} {{- $current := $page.Language.Lang -}} {{- $others := slice -}} {{- range hugo.Sites -}} {{- if ne .Language.Lang $current -}} {{- $others = $others | append . -}} {{- end -}} {{- end -}} {{- if $others }}
{{- range $others }} {{- $lang := .Language.Lang -}} {{- $label := or .Language.Label $lang -}} {{- with .Language.Params.label }}{{ $label = . }}{{ end -}} {{/* Find the matching translation of the current page */}} {{- $target := "" -}} {{- range $page.AllTranslations -}} {{- if eq .Language.Lang $lang -}} {{- $target = .Permalink -}} {{- end -}} {{- end -}} {{/* Fallback: the target language's home page */}} {{- if not $target }}{{ $target = .Home.RelPermalink }}{{ end }} {{ $label }} {{- end }}
{{- end }}
